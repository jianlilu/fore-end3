  7 Laro.register('perlin', function (La) {
  8 	this.start = true;
  9 	this.g1 = [];
 10 	this.p = [];
 11 		
 12 	this.noise = function (arg) {
 13 		var bx0, bx1,
 14 			rx0, rx1,
 15 			sx , t, u, v;
 16 
 17 		if (this.start) {
 18 			this.start = false;
 19 			this.init();
 20 		}
 21 
 22 		var s = this.setup(arg, bx0, bx1, rx0, rx1);
 23 
 24 		sx = this.s_curve(s.rx0);
 25 		u = s.rx0 * this.g1[this.p[s.bx0]];
 26 		v = s.rx1 * this.g1[this.p[s.bx1]];
 27 
 28 		return La.geometry.util.lerp(u, v, sx);
 29 	};
 30 	// ������ߣ��ɵ���
 31 	this.s_curve = function (t) {
 32 		return t * t * (3 - 2*t);
 33 	};
 34 
 35 	this.setup = function (i, bx0, bx1, rx0, rx1) {
 36 		var B = 0x100, // 256
 37 			BM = 0xff, // 255
 38 			N = 0x1000,// 4096
 39 			NP = 12,   // 
 40 			NM = 0xfff,// 4095
 41 			s = {};
 42 
 43 		s.t = i + N;
 44 		s.bx0 = Math.floor(s.t) & BM;
 45 		s.bx1 = (s.bx0 + 1) & BM;
 46 		s.rx0 = s.t - Math.floor(s.t);
 47 		s.rx1 = s.rx0 - 1;
 48 		return s;
 49 	};	
 50 
 51 	this.init = function () {
 52 		var B = 0x100, // 256
 53 			i, j, k;
 54 		for (i = 0; i < B; i ++) {
 55 			this.p[i] = i;
 56 			this.g1[i] = (Math.random() * (2*B) - B) / B;
 57 		}
 58 
 59 		while (-- i) {
 60 			k = this.p[i];
 61 			j = Math.floor(Math.random() * B);
 62 			this.p[i] = this.p[j];
 63 			this.p[j] = k;
 64 		}
 65 
 66 		for (i = 0; i < B + 2; i ++) {
 67 			this.p[B+i] = this.p[i];
 68 			this.g1[B + i] = this.g1[i];
 69 		}
 70 	}
 71 	
 72 })
