  6 	var Class = La.base.Class,
  7 		assert = La.err.assert,
  8 		toType = La.toType,
  9 		Pixel32 = La.geometry.Pixel32;

 11 	/**
 12 	 * ��ͼ����Դ

 16 	var Atlas = Class(function (obj) {
 17 		this.name = obj.name;
 18 		this.filename = obj.filename;
 19 		this.sources = {};
 20 
 21 		for (var k in obj.sources) {
 22 			if (obj.sources.hasOwnProperty(k)) {
 23 				var s = obj.sources[k];
 24 				this.sources[s.name] = s.data;
 25 			}
 26 		}
 27 	});

 29 	var checkFile = function (s) {
 30 		var suffix = s.lastIndexOf('.');
 31 		suffix = s.substr(suffix);
 32 		if (/png|jpg|jpeg|gif|bmp/.test(suffix)) {
 33 			return 'image';
 34 		} else if (/ogg|mp3|m4a|wav/.test(suffix)) {
 35 			return 'sound';
 36 		}
 37 	}
 38 	var isImage = function (s) {
 39 		return (checkFile(s) == 'image');
 40 	}
 41 	var isSound = function (s) {
 42 		return (checkFile(s) == 'sound');
 43 	}

 45 	/**
 46 	 * ResourceLoader
 47 	 * {Class}
 48 	 * ��Ҫ��image��loader
 49 	 */
 50 	var ResourceLoader = Class(function (path) {
 51 		// imagePath ��Դ��·����������
 52 		this.imagePath = path || 'resources/';
 53 		this.basePath = this.imagePath;
 54 		this.loadedImages = {};	
 55 		this.loadedSounds = {};
 56 
 57 	}).methods({
 58 
 59 		// ����һ��ͼƬ
 60 		loadImage: function (fileName) {
 61 			var image = this.loadedImages[fileName];
 62 			if (!!image) {
 63 				return image;
 64 			}
 65 
 66 			image = new Image();
 67 			image.src = this.imagePath + fileName;
 68 			this.loadedImages[fileName] = image;
 69 			return image;
 70 		},
 71 		// Ԥ���ض���ͼƬ
 72 		preloadImages: function (fileNames, callback) {
 73 			var imagesLoaded = 0,
 74 				_fileNames = [],
 75 				ind = -1,
 76 				args = arguments;
 77 
 78 			if (toType(fileNames) == 'array') {
 79 				_fileNames = fileNames;
 80 			} else {
 81 				while(toType(args[++ind]) == 'string') {
 82 					_fileNames.push(args[ind]);
 83 				}
 84 				callback = args[ind];
 85 			}
 86 
 87 			var length = _fileNames.length;
 88 			var callCallback = function (arg) {
 89 				!!callback && callback(arg);
 90 			};
 91 			var imageLoaded = function () {
 92 				imagesLoaded ++;
 93 				// ��һ�� �Ѽ��� �� ��Դ����ı����ȥ���Ա������ؽ��
 94 				callCallback(imagesLoaded/length);
 95 			};
 96 			var imageError = function () {
 97 				console.log('an image load error');
 98 			};
 99 
100 			for (var i = 0; i < length; i ++) {
101 				var fileName = _fileNames[i],
102 					image = this.loadedImages[fileName];
103 				if (image != undefined) {
104 					imagesLoaded ++;
105 					continue;
106 				}
107 
108 				image = new Image();
109 				image.src = this.imagePath + fileName;
110 				image.onload = imageLoaded;
111 				image.onerror = imageError;
112 				this.loadedImages[fileName] = image;
113 			}
114 
115 			callCallback(imagesLoaded / length);
116 		},
117 		// Ԥ���ض����Դ������ͼƬ�����ֵ�
118 		preload: function (files, callback) {
119 			var fileLoaded = 0,
120 				_fileNames = [],
121 				ind = -1,
122 				args = arguments;
123 
124 			if (toType(files) == 'array') {
125 				_fileNames = files;
126 			} else {
127 				while(toType(args[++ind]) == 'string') {
128 					_fileNames.push(args[ind]);
129 				}
130 				callback = args[ind];
131 			}
132 
133 			var length = _fileNames.length;
134 			var callCallback = function (arg) {
135 				!!callback && callback(arg);
136 			}
137 			var fileLoadedCB = function () {
138 				fileLoaded ++;
139 				callCallback(fileLoaded/length);
140 			};
141 			
142 			for (var i = 0; i < length; i ++) {
143 				// �ȼ���Ƿ��Ѿ�����
144 				var filename = _fileNames[i];
145 				if (isImage(filename)) {
146 					var image = this.loadedImages[filename];
147 					if (!!image) {
148 						fileLoaded ++;
149 						continue;
150 					}
151 					image = new Image();
152 					image.src = this.basePath + filename;
153 					image.onload = fileLoadedCB;
154 					image.onerror = fileLoadedCB;
155 					this.loadedImages[filename] = image;
156 				} else if (isSound(filename)) {
157 					var sound = this.loadedSounds[filename];
158 					if (!!sound) {
159 						fileLoaded ++;
160 						continue;
161 					}
162 					sound = new La.Sound(this.basePath + filename, fileLoadedCB);
163 					this.loadedSounds[filename] = sound;
164 				}
165 			}
166 
167 			callCallback(fileLoaded/length);
168 		}
169 	}).statics({
170         //��ȡ��ǰʵ��
171 		getInstance: function () {
172 			if (ResourceLoader.instance === null) {
173 				ResourceLoader.instance = new ResourceLoader(this.imagePath);
174 			}
175 			return ResourceLoader.instance;
176 		},
177 		instance: null	
178 	});

189 	// Game Font define
190 	var Font = Class(function (f) {
191 		this.baseColor = new Pixel32(f.base_r, f.base_g, f.base_b);
192 		this.outlineColor = new Pixel32(f.outline_r, f.outline_g, f.outline_b);
193 		this.size = f.size;
194 		this.font = f.font;
195 		this.outline = f.outline;
196 	}).methods({
197 		getFont: function () {
198 			var ret = [];
199 			ret.push('normal');
200 			ret.push(this.size + 'px');
201 			ret.push(this.font);
202 			return ret.join(' ');
203 		},
204 		/**
205 		 * ���һ����ʾ���ֵ�canvas
206 		 * @param text {String}
207 		 * @param wrapWidth {Number} һ�����ֿ�ȣ���������
208 		 */
209 		generateCanvas: function (text, wrapWidth) {
210 			var canvas = document.createElement('canvas'),
211 				c = canvas.getContext('2d');
212 			c.font = this.getFont();
213 
214 			var outline = this.outline * 2;
215 			if (wrapWidth != undefined) {
216 				text = this.wrapText(c, text, wrapWidth);
217 				canvas.width = c.measureText(text).width + 8 + outline * 2; // Ĭ�ϵ����ֶ�Ҫ��һ����margin
218 				canvas.height = (this.size + outline*4) * text.split('\n').length;
219 			} else {
220 				canvas.width = c.measureText(text).width + 8 + outline * 2;
221 				canvas.height = this.size + outline * 4;
222 			}
223 
224 			c.fillStyle = this.baseColor.toString();
225 			c.textBaseline = 'middle';
226 			c.textAlign = 'left';
227 			c.font = this.getFont();
228 
229 			if (outline != 0) {
230 				c.strokeStyle = this.outlineColor.toString();
231 				c.lineWidth = outline;
232 				c.strokeText(text, outline, this.size/2 + outline * 2);
233 			}
234 
235 			c.fillText(text, outline, this.size/2 + outline*2);
236 			return canvas;
237 		},
238 		wrapText: function (c, text, width) {
239 			var result = '',
240 				parts = text.split(' '),
241 				i = 0;
242 			while (i < parts.length) {
243 				if (c.measureText(result + parts[i]).width >= width) {
244 					result += '\n' + parts[i];
245 				} else if (i != 0) {
246 					result += ' ' + parts[i];
247 				} else {
248 					result += parts[i];
249 				}
250 				i ++;
251 			}
252 			return result;
253 		}
254 	});
