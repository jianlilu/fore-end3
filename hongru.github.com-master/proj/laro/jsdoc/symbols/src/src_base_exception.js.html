 12 /** 
 13  * @description
 14  * Package: Laro.err
 15  */
 16 
 17 Laro.register('.err', function (La) {
 18 	
 19 	/**
 20      * RuntimeException类
 21      * @class 
 22 
 23 	 * @memberOf Laro
 24      * @name RuntimeException
 25      * @constructor
 26      * 
 27      * @param {String} msg: 错误信息
 28 
 29      * @return RuntimeException类 实例
 30      */
 31 	/* runtime Error 扩展 */
 32 	function RuntimeException (msg) {
 33 		this.assign(msg);
 34 	}
 35 
 36 	RuntimeException.prototype = new Error();
 37 	RuntimeException.prototype.constructor = RuntimeException;
 38 	
 39 	/**
 40      * @lends Laro.RuntimeException.prototype
 41      */ 
 42 	RuntimeException.prototype.assign = function (msg) {
 43 		this.message = msg === undefined ? '' : msg;
 44 	};
 45 
 46 	/**
 47      * AssertionError
 48      * @class 
 49 
 50 	 * @memberOf Laro
 51      * @name AssertionError
 52      * @constructor
 53 	 * @extends Laro.RuntimeException
 54      * 
 55      * @param {String} msg: 错误信息
 56 
 57      * @return AssertionError 实例
 58      */
 59 	/* AssertionError */
 60 	function AssertionError (msg) {
 61 		this.assign(msg);
 62 	}
 63 
 64 	AssertionError.prototype = new RuntimeException();
 65 	AssertionError.prototype.constructor = AssertionError;
 66 
 67 	/**
 68      * Exception
 69      * @class 
 70 
 71 	 * @memberOf Laro
 72      * @name Exception
 73      * @constructor
 74 	 * @extends Laro.RuntimeException
 75      * 
 76      * @param {String} msg: 错误信息
 77 
 78      * @return Exception 实例
 79      */
 80 	/* Exception */
 81 	function Exception (msg) {
 82 		this.assign(msg);
 83 	}
 84 
 85 	Exception.prototype = new RuntimeException();
 86 	Exception.prototype.constructor = Exception;
 87 
 88 	
 89 	/**
 90      * 根据条件抛异常
 91      * 
 92      * @memberOf Laro
 93      * @function
 94      * 
 95      * @param {Boolean} condition: 抛异常的条件
 96      * @param {String} msg: 异常信息
 97      * @return AssertionError 实例
 98      */
 99 	/* interface */
100 	// 根据条件抛异常
101 	this.assert = function (condition, msg) {
102 		if (!condition) {
103 			throw new AssertionError(msg);
104 		}
105 	};
106 	this.RuntimeException = RuntimeException;
107 	this.AssertionError = AssertionError;
108 	this.Exception = Exception;
109     
110     // export to Laro
111     Laro.extend(this);
112 
113 })
