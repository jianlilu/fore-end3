 17 Laro.register('.game', function (La) {
 18     
 19     var Class = La.base.Class || La.Class;
 20     
 21     /**
 22      * Provides requestAnimationFrame in a cross browser way.
 23 	 * @function
 24 	 * @name requestAnimFrame
 25      */
 26     window.requestAnimFrame = this.requestAnimFrame =  (function() {
 27       return window.requestAnimationFrame ||
 28              window.webkitRequestAnimationFrame ||
 29              window.mozRequestAnimationFrame ||
 30              window.oRequestAnimationFrame ||
 31              window.msRequestAnimationFrame ||
 32              function(/* function FrameRequestCallback */ callback, /* DOMElement Element */ element) {
 33                window.setTimeout(callback, 1000/60);
 34              };
 35     })();
 36     
 37 	/**
 38      * 循环Loop类
 39      * @class 
 40      * @name Loop
 41      * @memberOf Laro
 42      * 
 43      * @param {Function} callback: 循环每帧的调用函数
 44      * @param {Object} that: 上下文对象
 45 
 46      * @return Loop 实例
 47      */
 48     var Loop = Class(function (callback, that) {
 49         var keepUpdating = true,
 50             lastLoopTime = new Date();
 51             
 52         function loop () {
 53             if (!keepUpdating) { return }
 54             requestAnimFrame(loop);
 55             
 56             var time = new Date(),
 57                 dt = (time - lastLoopTime) / 1000;
 58             // 时间间隔太长(大于3秒)，强制拉回dt
 59             if (dt >= 3) {
 60                 dt = 0.25;
 61             }
 62             
 63             callback.call(that, dt);
 64             lastLoopTime = time;
 65         }
 66         
 67         // 停止
 68         this.stop = function () {
 69             keepUpdating = false;
 70         }
 71         
 72         // 继续
 73         this.resume = function () {
 74             keepUpdating = true;
 75             lastLoopTime = new Date();
 76             loop();
 77         }
 78         
 79         loop();
 80         return this;
 81     });
 82     
 83     //screen fade or in
 84 	/**
 85      * ScreenTransitionFade - 渐变动画类
 86      * @class 
 87      * @name ScreenTransitionFade
 88      * @memberOf Laro
 89      * 
 90      * @param {Object} from: 渐变开始的 Pixel32 实例
 91      * @param {Object} to: 渐变结束的 Pixel32 实例
 92 	 * @param {Number} maxTime: 持续时间
 93 
 94      * @return ScreenTransitionFade 实例
 95      */
 96     var ScreenTransitionFade = Class(function (from, to, maxTime) {
 97         this.maxTime = maxTime;
 98         this.from = from;
 99         this.to = to;
100         
101         this.time = 0;
102         this.isDone = false;
103     }).methods({
104 	/**
105      * @lends Laro.ScreenTransitionFade.prototype
106      */
107 		/**
108 		 * 渐变动画更新方法
109 		 */
110         update: function (dt) {
111             this.time = Math.min(this.time + dt, this.maxTime);
112         },
113 		/**
114 		 * 渐变动画绘制方法
115 		 */
116         draw: function (render) {
117             this.isDone = this.time == this.maxTime;
118             var color = new La.Pixel32(La.lerp(this.from.r, this.to.r, this.time/this.maxTime),
119                                         La.lerp(this.from.g, this.to.g, this.time/this.maxTime),
120                                         La.lerp(this.from.b, this.to.b, this.time/this.maxTime),
121                                         La.lerp(this.from.a, this.to.a, this.time/this.maxTime));
122             color.a > 0 && render.drawFillScreen(color);
123         },
124 		/**
125 		 * 渐变动画重置
126 		 */
127         reset: function () {
128             this.time = 0;
129             this.isDone = false;
130         }
131     })
132     
133     this.Loop = Loop;
134     this.ScreenTransitionFade = ScreenTransitionFade;
135     
136     Laro.extend(this);
137 
138 })
