 12 /**
 13  * FSM 有限状态机
 14  * @description
 15  */ 
 16 
 17 Laro.register('.game', function (La) {
 18 		
 19 	var Class = La.Class || La.base.Class,
 20 		SimpleState = La.SimpleState || La.game.SimpleState,
 21 		BaseState = La.BaseState || La.game.BaseState;
 22    /**
 23      * FSM (状态机)类
 24      * @class 
 25      * @name FSM
 26      * @memberOf Laro
 27      * 
 28      * @param {Object} host: 状态机context（状态宿主）
 29      * @param {Array} states: 状态列表
 30      * @param {Function} onStateChange: 状态改变触发事件
 31 
 32      * @return FSM 实例
 33      */
 34 	var FSM = Class(function (host, states, onStateChange){
 35 		if (host == undefined) return;
 36 		this.host = host;
 37 		this.onStateChange = onStateChange;
 38 		this.stateArray = [];
 39 
 40 		// states list 中，俩个元素为一组，分别是state标识{Int,标识状态顺序},和对应的state 类
 41 		for (var i = 0; i < states.length; i += 2) {
 42 			var stateId = states[i],
 43 				stateClass = states[i + 1];
 44 
 45 			if (stateClass instanceof SimpleState) {
 46 				this.stateArray[stateId] = stateClass;
 47 			} else {
 48 				this.stateArray[stateId] = new stateClass(host, this, stateId);
 49 			}
 50 		}
 51 
 52 		this.currentState = FSM.kNoState;
 53 		this.numSuspended = 0;
 54 		this.suspendedArray = [];
 55 		this.numPreloaded = 0;
 56 		this.preloadedArray = [];
 57 		this.numStates = this.stateArray.length;
 58 
 59 	}).methods({
 60 	/**
 61      * @lends Laro.FSM.prototype
 62      */
 63 
 64 		/**
 65 		 * 进入某一个state
 66 		 * @param {String} startState: 状态名
 67 		 * @param {String} message: 进入状态的传递消息
 68 		 */
 69 		enter: function (startState, message) {
 70 			this.setState(startState, message);
 71 		},
 72 
 73 		/**
 74 		 * 退出state，回到初始化状态
 75 
 76 		 */
 77 		leave: function () {
 78 			this.setState(FSM.kNoState);
 79 		},
 80 
 81 		/**
 82 		 * 每帧状态机更新
 83 		 * @param {Number} dt: 主循环时间间隔
 84 		 */
 85 		update: function (dt) {
 86 			for (var i = 0; i < this.numSuspended; i ++) {
 87 				this.stateArray[this.suspendedArray[i]].suspended(dt);
 88 			}
 89 
 90 			if (this.currentState != FSM.kNoState) {
 91 				this.stateArray[this.currentState].update(dt);
 92 				// update 之后再判断transition
 93 				if (this.currentState != FSM.kNoState) {
 94 					this.stateArray[this.currentState].transition();
 95 				}
 96 			}
 97 		},
 98 		/**
 99 		 * 基于当前状态发出消息
100 		 * @param {String} msg: 消息
101 		 */
102 		message: function (msg) {
103 			this.currentState != FSM.kNoState && this.stateArray[this.currentState].message(msg);	 
104 		},
105 		/**
106 		 * 挂起某条消息
107 		 * @param {String} msg: 消息
108 		 */
109 		messageSuspended: function (msg) {
110 			for (var i = 0; i < this.numSuspended; i ++) {
111 				this.stateArray[this.suspendedArray[i]].message(msg);
112 			}				  
113 		},
114 		/**
115 		 * 尝试改变状态,根据一个判断来决定是否改变状态进入下一个状态
116 		 * @param {Boolean} condition: 是否改变状态的条件
117 		 * @param {String} toState: 目标状态的状态名
118 		 * @param {String} msg: 进入目标状态的消息
119 		 * @param {Boolean} reEnter: 是否可以重新进入当前状态
120 		 * @param {Boolean} suspendedCurrent: 是否挂起当前状态
121 		 * @return {Boolean} 尝试改变状态是否成功
122 		 */
123 		tryChangeState: function (condition, toState, msg, reEnter, suspendedCurrent) {
124 			if (reEnter == undefined) { reEnter = true } //重进当前状态
125 			if (suspendedCurrent == undefined) { suspendedCurrent = true }
126 			if (toState == FSM.kNextState) { toState = this.currentState + 1 }
127 
128 			if (condition 
129 				&& (toState != this.currentState || reEnter)) { console.log(toState)
130 				this.setState(toState, msg, suspendedCurrent);
131 				return true;
132 			}
133 
134 			return false;
135 		},
136 		/**
137 		 * 设置状态
138 		 * @param {String} state: 目标状态的状态名
139 		 * @param {String} msg: 进入目标状态的消息
140 		 * @param {Boolean} suspendedCurrent: 是否挂起当前状态
141 		 */
142 		setState: function (state, msg, suspendedCurrent) {
143 			if (state == FSM.kNextState) {
144 				state = this.currentState + 1;
145 			}
146 
147 			if (state == FSM.kNoState) {
148 				// 当前挂起的状态全部推出
149 				for ( ; this.numSuspended > 0; this.numSuspended --) {
150 					this.stateArray[this.suspendedArray[this.numSuspended - 1]].leave();
151 					this.stateArray[this.suspendedArray[this.numSuspended - 1]].isSuspended = false;
152 				}
153 				// 等待中的状态也全部终止
154 				for ( ; this.numPreloaded > 0; this.numPreloaded --) {
155 					this.stateArray[this.preloadedArray[this.numPreloaded - 1]].cancelPreload();
156 				}
157 			} else {
158 				if (suspendedCurrent) { // 需要挂起当前的状态
159 					this.stateArray[this.currentState].suspended();
160 					this.stateArray[this.currentState].isSuspended = true;
161 					this.suspendedArray[this.numSuspended ++] = this.currentState;
162 				} else {
163 					// 推出当前状态，进入指定状态
164 					if (this.currentState != FSM.kNoState) {
165 						this.stateArray[this.currentState].leave();
166 					}
167 
168 					// 如果指定状态并没有挂起的话，需要把所有挂起的状态退出
169 					if (!this.stateArray[state].isSuspended) {
170 						for ( ; this.numSuspended > 0; this.numSuspended --) {
171 							this.stateArray[this.suspendedArray[this.numSuspended - 1]].leave();
172 							this.stateArray[this.suspendedArray[this.numSuspended - 1]].isSuspended = false;
173 						}
174 					}
175 				}
176 			}
177 
178 			// 处理等待中的状态，如果不是指定状态，都取消
179 			for (var p = 0; p < this.numPreloaded; p++) {
180 				this.preloadedArray[p] != state && this.stateArray[this.preloadedArray[p]].cancelPreload();
181 			}
182 			this.numPreloaded = 0;
183 			// 从当前状态到指定状态
184 			this.onStateChange != undefined && this.onStateChange(this.currentState, state, msg);
185 
186 			var lastState = this.currentState;
187 			this.currentState = state;
188 
189 			if (this.currentState != FSM.kNoState) {
190 				if (this.stateArray[this.currentState].isSuspended) {
191 					// 状态转变后，如果状态是挂起的，不能是最后一个
192 					this.stateArray[this.currentState].resume(msg, lastState);
193 					this.stateArray[this.currentState].isSuspended = false;
194 					-- this.numSuspended;
195 				} else {
196 					// 进入指定状态
197 					this.stateArray[this.currentState].enter(msg, lastState);
198 				}
199 			}
200 		},
201 
202 		/**
203 		 * 获取当前状态
204 		 * @return {Object} 当前状态的实例对象
205 		 */
206 		getCurrentState: function () {
207 			if (this.currentState == FSM.kNoState) return null;
208 			return this.stateArray[this.currentState];
209 		},
210 		/**
211 		 * 预载入状态
212 		 * @param {String} 状态名
213 		 */
214 		preload: function (state) {
215 			this.preloadedArray[this.numPreloaded ++] = state;		 
216 		},
217 		/**
218 		 * 判断状态是否挂起
219 		 * @param {String} 状态名
220 		 * @return {Boolean} 是否挂起
221 		 */
222 		isSuspended: function (state) {
223 			return this.stateArray[state].isSuspended;			 
224 		}
225 	
226 	}).statics({
227 		kNoState : -1, // 默认初始化状态码
228 		kNextState: -2 // 默认进入下一个状态的状态码
229 	});
230 
231 	/**
232 	 * app 的 FSM
233 	 * 添加一些简单的交互处理
234      * AppFSM 类
235      * @class 
236      * @name AppFSM
237      * @memberOf Laro
238 	 * @extends Laro.FSM
239      * 
240 
241      * @return AppFSM 实例
242 	 */
243 	var AppFSM = FSM.extend().methods({
244 	/**
245      * @lends Laro.AppFSM.prototype
246      */
247 		
248 		/**
249 		 * 状态机派发draw方法
250 		 * @param {Object} render: 渲染实例
251 		 */
252 		draw: function (render) {
253 			// 如果状态类支持 draw，调用draw来绘制相关内容
254 			for (var i = 0; i < this.numSuspended; i ++) {
255 				this.stateArray[this.suspendedArray[i]].draw(render);
256 			}
257 			var s = this.getCurrentState();
258 			!!s && s.draw(render);
259 		},
260 		onMouse: function (x, y, left, leftPressed) {
261 			// 鼠标事件
262 			var s = this.getCurrentState();
263 			!!s && s.onMouse(x, y, left, leftPressed);
264 		}
265 	})
266 
267 	this.FSM = FSM;
268 	this.AppFSM = AppFSM;
269 	Laro.extend(this);
270 		
271 })
