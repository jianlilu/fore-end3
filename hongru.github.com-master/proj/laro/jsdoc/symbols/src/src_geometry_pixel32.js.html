1 /**
  2  * Pixel32
  3  * 32位像素点
  4  */
  5 Laro.register('.geometry', function (La) {
  6 		
  7 	var assert = La.err.assert,
  8 		Class = La.base.Class;
  9 
 10 	var Pixel32 = Class({
 11 		initialize: function (r, g, b, a) {
 12 			assert(r >= 0 && r <= 255, 'Pixel32 wrong --> r');
 13 			assert(g >= 0 && g <= 255, 'Pixel32 wrong --> g');
 14 			assert(b >= 0 && b <= 255, 'Pixel32 wrong --> b');
 15 
 16 			this.r = r;
 17 			this.g = g;
 18 			this.b = b;
 19 			this.a = a === undefined ? 255 : a;
 20 
 21 			this.normalized = [r/255.0, g/255.0, b/255.0, this.a > 1 ? this.a/255.0 : this.a];
 22 		},
 23 		equal: function (pix) {
 24 			if (pix instanceof Pixel32) {
 25 				return this.r == pix.r 
 26 						&& this.g == pix.g
 27 						&& this.b == pix.b
 28 						&& this.a == pix.a;
 29 			} else {
 30 				return false;
 31 			}
 32 		},
 33 		toString: function () {
 34 			return 'rgba('+ this.r +', '+ this.g +', '+ this.b +', '+ this.normalized[3] +')';		  
 35 		},
 36 		rgbString: function () {
 37 			return 'rgb('+ this.r +', '+ this.g +', '+ this.b +')';
 38 		}
 39 	});
 40 
 41 	this.Pixel32 = Pixel32;
 42     
 43     Laro.extend(this);
 44 })
