1 /**
  2  * A* 寻路算法
  3  * 基于map为规则矩阵的形式
  4  */
  5 
  6 Laro.register('.tools', function (La) {
  7 	var pkg = this;
  8 
  9 	var P = function (parentP, pos, endPos) {
 10 		this.pos = pos;
 11 		this.endPos = endPos;
 12 		if (parentP) {
 13 			this.parentP = parentP;
 14 			this.parentPos = parentP.pos;
 15 		}
 16 		
 17 	};
 18 	P.prototype = {
 19 		getGByPoint: function (p) {
 20 			var retG = p.G || 0;
 21 			if (p.pos[0] == this.pos[0] || p.pos[1] == this.pos[1]) {
 22 				retG += 10;
 23 			} else {
 24 				retG += 14;
 25 			}
 26 			return retG;
 27 		},
 28 		updateH: function (endPos) {
 29 			if (!endPos) {
 30 				endPos = this.endPos;
 31 			}	 
 32 			this.H = 10*(Math.abs(this.pos[0]-this.endPos[0]) + Math.abs(this.pos[1]-this.endPos[1]));
 33 		},
 34 		updateF: function () {
 35 			this.F = this.G + this.H;		 
 36 		}
 37 
 38 	};
 39 
 40 	/**
 41 	 * 所有用于寻路算法的mapMatrix,这里都简化成可通过，和不可通过两种方式
 42 	 * 可通过统一用 0 ，不可通过统一 用 大于0的数字表示
 43 	 * 所以在使用以下方法时，可能需要先 将 你的mapMatrix 格式化一下
 44 	 */
 45 	var AStar = La.Class(function (mapMatrix, whiteList) {
 46 		this.mapMatrix = mapMatrix;
 47 		this.mmRow = mapMatrix.length;
 48 		this.mmCol = mapMatrix[0].length;
 49 		// 用于处理可以斜着穿透拐角的白名单，默认都是不可斜穿
 50 		if (whiteList == undefined) {
 51 			whiteList = [];
 52 		}
 53 		this.whiteList = whiteList;
 54 		this._whiteHash = {};
 55 		for (var i = 0; i < this.whiteList.length; i ++) {
 56 			this._whiteHash[this.whiteList[i]] = 1;
 57 		}
 58 		
 59 	}).methods({
 60 		// startPos,endPos 分别表示寻路相对于mapMatrix的起始位置 和 结束位置
 61 		// 比如第几行，第几列
 62 		getPath: function (startPos, endPos) {
 63 			this.startPos = startPos;
 64 			this.endPos = endPos;
 65 			this.currP = null;
 66 
 67 			this.openHash = {};
 68 			this.closeHash = {};
 69 
 70 			this.path = [];
 71 			
 72 			// whether startPos or endPos is out of mapMatrix, or not available;
 73 			if (this.mapMatrix[startPos[0]][startPos[1]] > 0
 74 				|| this.mapMatrix[startPos[0]][startPos[1]] == undefined
 75 				|| this.mapMatrix[endPos[0]][endPos[1]] > 0
 76 				|| this.mapMatrix[endPos[0]][endPos[1]] == undefined) {
 77 				return null;
 78 			} 
 79 
 80 			// push in openList
 81 			//this.openList.push(startPos);
 82 			var startP = new P(null, startPos, endPos);
 83 			this.openHash[startPos.join('_')] = startP;
 84 			this.currP = startP;
 85 			this._step();
 86 
 87 			return this.path;
 88 		},
 89 		_step: function () {
 90 			// step a
 91 			var F = Number.MAX_VALUE, isEmpty = true;
 92 			for (var k in this.openHash) {
 93 				isEmpty = false;
 94 				if (this.openHash[k].F < F) {
 95 					F = this.openHash[k].F;
 96 					this.currP = this.openHash[k];
 97 				}
 98 			}
 99 			if (isEmpty) {
100 				// not find path;
101 				return false;
102 			}
103 
104 			// step b
105 			var key = this.currP.pos.join('_');
106 			delete this.openHash[key];
107 			this.closeHash[key] = 1;
108 
109 			// step c
110 			var pos = this.currP.pos,
111 				endPos = this.endPos;
112 			for (var r = pos[0]-1; r <= pos[0]+1; r ++) {
113 				var row = this.mapMatrix[r];
114 				if (!!row) {
115 					for (var c = pos[1]-1; c <= pos[1]+1; c ++) {
116 						var _pos = [r, c];
117 						if (row[c] === 0) {
118 							// available;
119 							//in closeHash, continue;
120 							//or in corner rules, continue;
121 							if (this.closeHash[_pos.join('_')]
122 								|| (row[pos[1]] > 0 && !this._whiteHash[row[pos[1]]]) 
123 								|| (this.mapMatrix[pos[0]][c] > 0 && !this._whiteHash[this.mapMatrix[pos[0]][c]])) {
124 								continue;
125 							}
126 
127 							// not in openHash
128 							if (!this.openHash[_pos.join('_')]) {
129 								var p = new P(this.currP, _pos, endPos);
130 								p.G = p.getGByPoint(this.currP);
131 								p.updateH();
132 								p.updateF();
133 								this.openHash[p.pos.join('_')] = p;
134 							} else {
135 							// already in openHash
136 								var p = this.openHash[_pos.join('_')],
137 									eG = p.getGByPoint(this.currP);
138 								if (p.G > eG) {
139 									p.G = eG;
140 									p.updateF();
141 									p.parentP = this.currP;
142 									p.parentPos = this.currP.pos;
143 								}
144 							}
145 
146 						} else if (row[c] > 0) {
147 							this.closeHash[_pos.join('_')] = 1;
148 						}
149 					}
150 				}
151 			}
152 
153 			// step d
154 			// stop or not
155 			if (this.currP.pos[0] == this.endPos[0] && this.currP.pos[1] == this.endPos[1]) {
156 				// get path successfully
157 				this._composePath(this.currP);
158 				return true;
159 			} else {
160 				arguments.callee.call(this);
161 			}
162 
163 
164 		},
165 		// compose path
166 		_composePath: function (p) {
167 			this.path.unshift(p.pos);
168 			if (p.parentP) {
169 				arguments.callee.call(this, p.parentP);
170 			}
171 
172 		}
173 		
174 	});
175 
176 	this.AStar = AStar;
177 
178 })
