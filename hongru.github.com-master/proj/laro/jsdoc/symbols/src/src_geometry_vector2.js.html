  6 Laro.register('.geometry', function (La) {
  7 		
  8 	var self = this,
  9 		Class = La.base.Class,
 10 		Point2 = La.geometry.Point2;
 11 
 12 	var Vector2 = Point2.extend({
 13 		// ʸ�����
 14 		dot: function (v) {
 15 			return this.x * v.x + this.y * v.y;
 16 		},
 17 		
 18 		cross: function (v) {
 19 			return this.x * v.x - this.y * v.y; 
 20 		},
 21 		length: function () {
 22 			return this.magnitude();
 23 		},
 24 		// ��λ��
 25 		normalize: function () {
 26 			var inv = 1 / this.length();
 27 			this.x *= inv;
 28 			this.y *= inv;
 29 			return this;
 30 		},
 31 		copy: function () {
 32 			return new Vector2(this.x, this.y);
 33 		} 
 34 	});
 35 
 36 	Vector2.zero = new Vector2(0, 0);
 37 	Vector2.X = new Vector2(1, 0);
 38 	Vector2.Y = new Vector2(0, 1);
 39 
 40 	this.Vector2 = Vector2;
 41     
 42     Laro.extend(this);
 43 		
 44 })
