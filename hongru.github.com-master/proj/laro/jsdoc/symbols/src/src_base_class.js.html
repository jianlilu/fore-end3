 17 Laro.register('.base', function (L) {
 18 
 19     var context = this,
 20     old = context.Class,
 21     f = 'function',
 22     fnTest = /xyz/.test(function() {
 23         xyz
 24     }) ? /\bsupr\b/: /.*/,
 25     proto = 'prototype';
 26 	
 27 	/**
 28      * 创建一个类
 29      * 
 30      * @memberOf Laro
 31      * @function
 32      * 
 33      * @param {Function} o: 类的constructor， 也可以是一个Object字典，如果是这样，那么initialize默认为constructor，其他的methods
 34 	 * @example 
 35 	 * Man = Laro.Class(function (name) {
 36 			this.name = name;
 37 		}).methods({
 38 			walk: function () {
 39 				//todo
 40 			},
 41 			talk: function () {
 42 				//todo
 43 			}
 44 		});
 45 		
 46 		SuperMan = Man.extend(function (name, canfly) {
 47 			this.name = name;
 48 			this.canfly = canfly;
 49 		}).methods({
 50 			walk: function () {
 51 				this.supr();
 52 				// todo
 53 			},
 54 			fly: function () {
 55 				// todo
 56 			}
 57 		});
 58      * 
 59      * @return {Class} 返回一个类
 60      */
 61     function Class(o) {
 62         return extend.call(isFn(o) ? o: function() {},
 63         o, 1);
 64     }
 65 
 66     function isFn(o) {
 67         return typeof o === f;
 68     }
 69 
 70     function wrap(k, fn, supr) {
 71         return function() {
 72             var tmp = this.supr;
 73             this.supr = supr[proto][k];
 74             var ret = fn.apply(this, arguments);
 75             this.supr = tmp;
 76             return ret;
 77         }
 78     }
 79 
 80     function process(what, o, supr) {
 81         for (var k in o) {
 82             if (o.hasOwnProperty(k)) {
 83                 what[k] = isFn(o[k]) && isFn(supr[proto][k]) && fnTest.test(o[k]) ? wrap(k, o[k], supr) : o[k];
 84             }
 85         }
 86     }
 87 
 88 	/**
 89      * @lends Laro.Class(...)
 90 	 * @param {Function} o: 子类的constructor
 91 	 * @param {Boolean} fromSub: 不继承父类
 92 	 * @return {Class} 返回一个类的子类
 93      */ 
 94     function extend(o, fromSub) {
 95         // must redefine noop each time so it doesn't inherit from previous arbitrary classes
 96         function noop() {}
 97         noop[proto] = this[proto];
 98         
 99         var supr = this,
100         prototype = new noop(),
101         isFunction = isFn(o),
102         _constructor = isFunction ? o: this,
103         _methods = isFunction ? {}: o;
104         
105         function fn() {
106             if (this.initialize) this.initialize.apply(this, arguments);
107             else {
108                 fromSub || isFunction && supr.apply(this, arguments);
109                 _constructor.apply(this, arguments);
110             }
111         }
112 
113         fn.methods = function(o) {
114             process(prototype, o, supr);
115             fn[proto] = prototype;
116             return this;
117         }
118 
119         fn.methods.call(fn, _methods).prototype.constructor = fn;
120 
121         fn.extend = arguments.callee;
122         fn[proto].implement = fn.statics = function(o, optFn) {
123             o = typeof o == 'string' ? (function() {
124                 var obj = {};
125                 obj[o] = optFn;
126                 return obj;
127             } ()) : o;
128             process(this, o, supr);
129             return this;
130         }
131 
132         return fn;
133     }
134 
135     Class.noConflict = function() {
136         context.Class = old;
137         return this;
138     }
139     context.Class = Class;
140 
141     Laro.Class = Class;
142     
143 });
