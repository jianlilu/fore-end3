 17 Laro.register('.action', function (La) {
 18 	/**
 19      * 命名空间 Laro.action
 20      * 
 21      * @namespace
 22      * @name Laro.action
 23      */
 24 	var Class = La.Class || La.base.Class;
 25 
 26 	/**
 27      * 帧动画控制类
 28      * @class 
 29 
 30 	 * @memberOf Laro
 31      * @name AnimationHandle
 32      * @constructor
 33      * 
 34      * @param {Object} animation: Animation实例（包含Animation需要的各种信息）
 35      * @param {Function} callback: 帧动画播放回调函数
 36      * @param {Boolean} mirrored: 是否横向翻转
 37      * @return 帧动画控制实例
 38      */
 39 	var AnimationHandle = Class(function (animation, callback, mirrored) {
 40 		if (animation instanceof AnimationHandle) {
 41 			animation = animation.animation;
 42 		}
 43 		if (mirrored == undefined) {
 44 			mirrored = false;
 45 		}
 46 
 47 		this.animation = animation;
 48 		this.callback = callback == undefined ? null : callback;
 49 		this.currentFrame = 0;
 50 		this.time = 0;
 51 		this.renderMirrored = mirrored;
 52 
 53 		this.speed = 1;
 54 		this.start = 0;
 55 		this.end = 1;
 56 
 57 		this.playTo = -1;
 58 		this.loop = true;
 59 		this.playing = false;
 60 
 61 	}).methods({
 62 	/**
 63      * @lends Laro.AnimationHandle.prototype
 64      */ 
 65 	 
 66 		/**
 67 		 * 克隆一个 AnimationHandle 实例
 68 		 *
 69 		 * @return AnimationHandle 实例
 70 		 */
 71 		clone: function () {
 72 			var clone = new AnimationHandle(this.animation, this.callback, this.renderMirrored);
 73 			clone.start = this.start;
 74 			clone.end = this.end;
 75 			clone.time = this.time;
 76 			return clone;
 77 		},
 78 		/**
 79 		 * 用于帧动画更新的update方法
 80 		 * @param {Number} dt: 帧间隔时间
 81 		 */
 82 		update: function (dt) {
 83 			if (!this.playing) {
 84 				this.currentFrame = Math.floor(this.time * this.animation.framerate) % this.animation.nbrOfFrames;
 85 				return;
 86 			}
 87 
 88 			var oldTime = this.time;
 89 			this.time += this.speed * dt;
 90 
 91 			var halfFrame = 0.5/this.animation.framerate;
 92 			var animationLength = this.animation.animationLength;
 93 			
 94 			// 循环
 95 			if (this.loop) {
 96 				if (this.speed > 0) {
 97 					this.time = this.time >= animationLength * this.end ? this.start * animationLength : this.time;
 98 				} else {
 99 					this.time = this.time <= animationLength * this.start ? this.end * animationLength - halfFrame : this.time;
100 				}
101 			} else {
102 				var tempPlayTo; 
103 				if (this.speed > 0) {
104 					tempPlayTo = this.playTo >= 0 ? this.playTo : this.end;
105 					if (this.time >= animationLength * tempPlayTo) { 
106 						this.time = animationLength * tempPlayTo - halfFrame;
107 						this.playing = false;
108 					}
109 				} else {
110 					tempPlayTo = this.playTo >= 0 ? this.playTo : this.start;
111 					if (this.time <= animationLength * tempPlayTo) {
112 						this.time = animationLength * tempPlayTo + halfFrame;
113 						this.playing = false;
114 					}
115 				}
116 			}
117 
118 			this.time = Math.max(this.time, 0);
119 			this.currentFrame = Math.floor(this.time * this.animation.framerate) % this.animation.nbrOfFrames;
120 
121 			if (this.callback != null) {
122 				var timeToCheck = this.playing ? this.time : (this.speed > 0 ? animationLength * this.end : this.animation.animationLength * this.start);
123 				var evts;
124 				if (oldTime < timeToCheck) {
125 					evts = this.animation.getEvents(oldTime, timeToCheck);
126 				} else {
127 					evts = this.animation.getEventsSlow(oldTime, timeToCheck, animationLength * this.start, animationLength * this.end, dt);
128 				}
129 
130 				if (evts.length >= 2) {
131 					this.time = this.animation.getTimeForNextEvent(oldTime, timeToCheck);
132 					evts = [evts[0]];
133 				}
134 
135 				for (var e = 0; e < evts.length; e++) {
136 					this.callback(evts[e], this);
137 				}
138 
139 				if (!this.playing) {
140 					this.callback('stopped', this);
141 				}
142 			}
143 		},
144 		/**
145 		 * 帧动画绘制方法
146 		 * @param {Object} render: Laro.CanvasRender 实例
147 		 * @param {Number} x: 绘制在画布上的x坐标
148 		 * @param {Number} y: 绘制在画布上的y坐标
149 		 * @param {Number} angle: 绘制角度
150 		 * @param {Number} alpha: 绘制透明度
151 		 * @param {Color} tint: 边缘混合色
152 		 */
153 		draw : function (render, x, y, angle, alpha, tint) {
154 			var image = this.animation.frames[this.currentFrame];
155 			var baseX = this.renderMirrored ? x - (image.textureWidth - this.animation.pivotx) : x - this.animation.pivotx;
156 			render.drawImage(image, baseX, y - this.animation.pivoty, angle, false, alpha, tint, this.renderMirrored);
157 		},
158 		/**
159 		 * 帧动画横向翻转
160 		 */
161 		mirror: function () {
162 			this.renderMirrored = !this.renderMirrored;		
163 		},
164 		/**
165 		 * 播放帧动画
166 		 * @param {Boolean} loop: 是否循环，默认为true
167 		 */
168 		play: function (loop) {
169 			this.playTo = -1;
170 			if (loop == undefined) {
171 				loop = true;
172 			}
173 			if (this.time >= this.end * this.animation.animationLength - 0.5 / this.animation.framerate) {
174 				this.time = this.start * this.animation.animationLength;
175 			}
176 			this.loop = loop;
177 			this.playing = true;
178 		},
179 		/**
180 		 * 播放帧动画到指定时间点
181 		 * @param {Number} t: 指定时间点
182 		 */
183 		playToTime: function (t) {
184 			this.playTo = t;
185 			if (this.time >= this.playTo * this.animation.animationLength - 0.5 / this.animation.framerate) {
186 				this.time = this.start * this.animation.animationLength;
187 			}
188 			this.playing = true;
189 		},
190 		/**
191 		 * 播放帧动画到指定事件触发处
192 		 * @param {String} name: 事件名
193 		 */
194 		playToEvent: function (name) {
195 			for (var i = 0; i < this.animation.events.length; i ++) {
196 				var e = this.animation.events[i];
197 				if (e.name == name) {
198 					this.playToTime(e.time / this.animation.animationLength);
199 					break;
200 				}
201 			}			 
202 		},
203 		/**
204 		 * 停止帧动画
205 		 */
206 		stop: function () {
207 			this.playing = false;	  
208 		},
209 		/**
210 		 * 返回到开始位置
211 		 */
212 		rewind: function () {
213 			this.time = this.start * this.animation.animationLength;		
214 		},
215 		/**
216 		 * 跳到指定位置
217 		 * @param {Number} time: 指定时间点
218 		 */
219 		gotoTime: function (time) {
220 			this.time = time * this.animation.animationLength;		  
221 		},
222 		/**
223 		 * 跳到指定事件位置
224 		 * @param {String} evt: 指定事件名
225 		 */
226 		gotoEvent: function (evt) {
227 			for (var i = 0; i < this.animation.events.length; i ++) {
228 				var e = this.animation.events[i];
229 				if (e.name == evt) {
230 					this.time = e.time;
231 					break;
232 				}
233 			}		   
234 		},
235 		/**
236 		 * 跳到动画结束位置
237 		 */
238 		gotoEnd: function () {
239 			var halfFrame = 0.5 / this.animation.framerate;
240 			this.time = (this.end - halfFrame) * this.animation.animationLength;
241 		},
242 		/**
243 		 * 指定动画播放区间
244 		 * @param {Number} s: 指定区间开始
245 		 * @param {Number} e: 指定区间结束
246 		 */
247 		setRange: function (s, e) {
248 			this.start = s;
249 			this.end = e;
250 
251 			var length = this.animation.animationLength;
252 			if (this.time < s * length) this.time = s * length;
253 			if (this.time > e * length) this.time = e * length;
254 		},
255 		/**
256 		 * 指定动画播放速度
257 		 * @param {Number} s: 播放速度
258 		 */
259 		setSpeed: function (s) {
260 			this.speed = s;		  
261 		},
262 		/**
263 		 * 获取用于播放的动画长度
264 		 */
265 		getLength: function () {
266 			return this.animation.animationLength * (this.end - this.start);		   
267 		},
268 		/**
269 		 * 获取当前位置
270 		 */
271 		getCurrentPosition: function () {
272 			return this.time;					
273 		},
274 		/**
275 		 * 是否停止
276 		 */
277 		isStopped: function () {
278 			return !this.playing;		   
279 		},
280 		/**
281 		 * 设置播放回调
282 		 * @param {Function} cb: 播放回调
283 		 */
284 		setCallback: function (cb) {
285 			this.callback = cb;			 
286 		}
287 	});
288 
289 	this.AnimationHandle = AnimationHandle;
290 	Laro.extend(this);
291 		
292 })
