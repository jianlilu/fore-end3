  6 Laro.register('.geometry', function (La) {
  7 
  8 	this.Circle = function Circle(cent, rad) {
  9 		this.c = cent;
 10 		this.r = rad;
 11 	};
 12 	La.Circle = this.Circle;
 13 		
 14 	var Class = La.base.Class || La.Class,
 15 		assert = La.err.assert,
 16 		Vector2 = La.geometry.Vector2,
 17 		Point2 = La.geometry.Point2,
 18 		Circle = La.geometry.Circle || this.Circle;
 19 
 20 	var Rectf = Class(function (x0, y0, x1, y1) {
 21 		this.x0 = x0;
 22 		this.x1 = x1;
 23 		this.y0 = y0;
 24 		this.y1 = y1;
 25 		this.width = x1 - x0 + 1;
 26 		this.height = y1 - y0 + 1;
 27 	}).methods({
 28 		// �������ģ�����һ������
 29 		center: function () {
 30 			return new Vector2((this.x0 + this.x1 + 1)/2, (this.y0 + this.y1 + 1)/2);
 31 		},
 32 		// ��ָ���ᷭת
 33 		invertBy: function (f) {
 34 			f = f || 'x';
 35 			if (f == 'x') {
 36 				return new Rectf(-this.x1, this.y0, -this.x0, this.y1);
 37 			} else if (f == 'y') {
 38 				return new Rectf(this.x0, -this.y1, this.x1, -this.y0);
 39 			}
 40 		},
 41 		// �ƶ�
 42 		offset: function (x, y) {
 43 			if (x instanceof Vector2 || x instanceof Point2) {
 44 				x = x.x;
 45 				y = x.y;
 46 			}
 47 			return new Rectf(this.x0 + x, this.y0 + y, this.x1 + x, this.y1 + y);
 48 		},
 49 		// ��չ����
 50 		expand: function (w, h) {
 51 			if (w instanceof Vector2 || w instanceof Point2) {
 52 				w = w.x;
 53 				h = w.y;
 54 			}
 55 			return new Rectf(this.x0 - w, this.y0 - h, this.x1 + w, this.y1 + h);
 56 		},
 57 		// ����Ƿ��
 58 		// ֧�� �Ե㣬 ������ ���εļ��
 59 		contains: function (x, y) {
 60 			var o = x;
 61 			if (x instanceof Vector2 || x instanceof Point2) {
 62 				// ��ָ���������İ��ϵ
 63 				return this.x0 <= o.x
 64 					&& this.y0 <= o.y
 65 					&& this.x1 >= o.x
 66 					&& this.y1 >= o.y;
 67 			} else if (x instanceof Rectf) {
 68 				// ���ΰ��ϵ���
 69 				return this.x0 <= o.x0
 70 					&& this.y0 <= o.y0
 71 					&& this.x1 >= o.x1
 72 					&& this.y1 >= o.y1;
 73 			} else {
 74 				// �Զ��� x y ���ĵ�
 75 				return this.x0 <= x
 76 					&& this.y0 <= y
 77 					&& this.x1 >= x
 78 					&& this.y1 >= y;
 79 			}
 80 		},
 81 		// �����ж�
 82 		// ֧�� Բ�Σ� ����
 83 		overlaps: function (shape, ox, oy) {
 84 			var r, x, y;
 85 			if (ox == undefined && oy == undefined) {
 86 				ox = 0;
 87 				oy = 0;
 88 			} else if (ox instanceof Vector2 || ox instanceof Point2) {
 89 				ox = ox.x;
 90 				oy = ox.y;
 91 			}
 92 
 93 			if (shape instanceof Circle) {
 94 				r = shape.r;
 95 				// Բ��λ��
 96 				x = shape.c.x + ox;
 97 				y = shape.c.y + oy;
 98 				return this.x0 - r <= x
 99 					&& this.y0 - r <= y
100 					&& this.x1 + r >= x
101 					&& this.y1 + r >= y;
102 			} else if (shape instanceof Rectf) {
103 				return !(this.x0 > shape.x1 + ox || this.x1 < shape.x0 + ox || this.y0 > shape.y1 + oy || this.y1 < shape.y0 + oy);
104 			} else {
105 				// TODO
106 			}
107 		},
108 		// �þ���ȥ����һ������
109 		clip: function (v) {
110 			return new Vector2(Math.max(this.x0, Math.min(this.x1, v.x)), Math.max(this.y0, Math.min(this.y1, v.y)));	   
111 		},
112 		// copy
113 		copy : function () {
114 			return new Rectf(this.x0, this.y0, this.x1, this.y1);
115 		},
116 		// include
117 		include: function (x, y) {
118 			if (x instanceof Vector2 || x instanceof Point2) {
119 				x = x.x;
120 				y = x.y;
121 			}
122 			this.x0 = Math.min(this.x0, x);
123 			this.y0 = Math.min(this.y0, y);
124 			this.x1 = Math.max(this.x1, x);
125 			this.y1 = Math.max(this.y1, y);
126 			this.width = this.x1 - this.x0 + 1;
127 			this.height = this.y1 - this.y0 + 1;
128 		}
129 	});

132 	this.Rectangle = Rectf;
