  5 Laro.register('.game', function (La) {
  6 		
  7 	var Class = La.Class || La.base.Class,
  8 		Pixel32 = La.Pixel32 || La.geometry.Pixel32,
  9 		toType = La.toType;
 10 
 11 	// ����state��ķ��� ����Ҫ ������д����á�
 12 	var BaseState = Class(function (host, fsm, id) {
 13 		if (host == undefined || toType(host) != 'object') return;
 14 		this.host = host;
 15 		this.fsm = fsm;
 16 		this.stateId = id;
 17 		this.isSuspended = false;
 18 			
 19 	}).methods({
 20 		enter: function (msg, fromState) {throw 'no enter'},
 21 		leave: function () {throw 'no leave'},
 22 		update: function (dt) {throw 'no update'},
 23 		suspended: function (dt) {throw 'no suspended'},
 24 		message: function (msg) { throw 'no message'},
 25 		suspend: function () {throw 'no suspend'},
 26 		resume: function (msg, fromState) {throw 'no resume'},
 27 		preload: function () {throw 'no preload'},
 28 		cancelPreload: function () {throw 'no cancelPreload'},
 29 		transition: function () { return false }
 30 		
 31 	});
 32 
 33 	/**
 34 	 * SimpleState
 35 	 */
 36 	var SimpleState = BaseState.extend(function (id, enterFn, leaveFn, updateFn) {
 37 		this.stateId = id;
 38 		var emptyFn = function () {};
 39 		this.isSuspended = false;
 40 	
 41 		this.enter = enterFn != null ? enterFn : emptyFn;
 42 		this.leave = leaveFn != null ? leaveFn : emptyFn;
 43 		this.update = updateFn != null ? updateFn : emptyFn;
 44 	});
 45 
 46 	/**
 47 	 * AppState
 48 	 */
 49 	var AppState = BaseState.extend(function () {
 50 		this.isSuspended = false;
 51 		this.dimTimeLeft = 0;
 52 	}).methods({
 53 		update: function (dt) {
 54 			this.dimTimeLeft -= dt;
 55 			if (this.dimTimeLeft < 0) this.dimTimeLeft = 0;
 56 		},
 57 		draw: function (render) {
 58 			if (this.dimTimeLeft > 0) {
 59 				var color = new Pixel32(0, 0, 0, Math.min(255, 765*this.dimTimeLeft));
 60 				render.drawQuad(null, color);
 61 			}
 62 		},
 63 		suspended: function (dt) {
 64 			this.dimTimeLeft = 0.25;
 65 		},
 66 		onMouse: function (x, y, left, leftPressed) { throw 'no onMouse' }
 67 		
 68 	});
 69 
 70 	this.BaseState = BaseState;
 71 	this.SimpleState = SimpleState;
 72 	this.AppState = AppState;
 73 		
 74 })
