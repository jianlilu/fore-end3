 86 		drawImage: function (imgW, x, y, angle, centered, alpha, tint, hFlipped) {
 87 			if (this.calls++ > this.maxCalls) return;
 88 			this.context.save();
 89 			if (toType(alpha) == 'number' && alpha != 1) {

115 			}
116 			// offset of the image
117 			this.context.translate(imgW.offsetX, imgW.offsetY);
118 			x += imgW.offsetX;
119 			y += imgW.offsetY;
120 
121 			if(!this.frontToBack) {

168 			} else {

170 			}

278 		drawTriangleImage : function(image, xy, uv, tint) {
279 			var x0 = xy[0];
280 			var y0 = xy[1];
281 			var x1 = xy[2];
282 			var y1 = xy[3];
283 			var x2 = xy[4];
284 			var y2 = xy[5];
285  
286 			var sx0 = uv[0];
287 			var sy0 = uv[1];
288 			var sx1 = uv[2];
289 			var sy1 = uv[3];
290 			var sx2 = uv[4];
291 			var sy2 = uv[5];
292  
293 			this.context.save();
294 			this.context.beginPath();
295 			this.context.moveTo(x0, y0);
296 			this.context.lineTo(x1, y1);
297 			this.context.lineTo(x2, y2);
298 			this.context.closePath();
299 			this.context.clip();
300  
301 			var denom = sx0 * (sy2 - sy1) - sx1 * sy2 + sx2 * sy1 + (sx1 - sx2) * sy0;
302 			if (denom === 0) return;
303  
304 			var m11 = -(sy0 * (x2 - x1) - sy1 * x2 + sy2 * x1 + (sy1 - sy2) * x0) / denom;
305 			var m12 = (sy1 * y2 + sy0 * (y1 - y2) - sy2 * y1 + (sy2 - sy1) * y0) / denom;
306 			var m21 = (sx0 * (x2 - x1) - sx1 * x2 + sx2 * x1 + (sx1 - sx2) * x0) / denom;
307 			var m22 = -(sx1 * y2 + sx0 * (y1 - y2) - sx2 * y1 + (sx2 - sx1) * y0) / denom;
308 			var dx = (sx0 * (sy2 * x1 - sy1 * x2) + sy0 * (sx1 * x2 - sx2 * x1) + (sx2 * sy1 - sx1 * sy2) * x0) / denom;
309 			var dy = (sx0 * (sy2 * y1 - sy1 * y2) + sy0 * (sx1 * y2 - sx2 * y1) + (sx2 * sy1 - sx1 * sy2) * y0) / denom;
310  
311 			this.context.transform(m11, m12, m21, m22, dx, dy);
312 			this.drawEMBImage(image, 0, 0, false, this.context);
313 			this.context.restore();

334 		drawLayer: function (layer, ox, oy, x, y, nx, ny) {
335 			var i, j, offset;
336 			if (layer instanceof TileLayer) {
337 				for (j = y; j < y + ny; j ++) {
338 					var previous = layer.previous(x, j),
339 						last = layer.index(x + nx - 1, j);
340 					var img, ix, iy, centered, flipped;
341 					for (i = previous + 1, offset = i * 5; i <= last; i ++) {
342 						if (this.calls ++ > this.maxCalls) continue;
343 						img = layer.tiles[offset++];
344 						ix = layer.tiles[offset++];
345 						iy = layer.tiles[offset++];
346 						offset ++;
347 						flipped = layer.tiles[offset++];
348 						// draw tile image
349 						var px = ox + ix + img.offsetX,
350 							py = oy + iy + img.offsetY,
351 							width = img.width,
352 							height = img.height;
353 
354 						if (this.scaleFactor != 1) {
355 							px = this.scale(px);
356 							py = this.scale(py);
357 							width = this.scale(width);
358 							height = this.scale(height);
359 						}
360 
361 						if (!flipped) {
362 							this.context.drawImage(img.image, img.x, img.y, img.width, img.height, px, py, width, height);
363 						} else {
364 							this.context.scale(-1, 1);
365 							this.context.drawImage(img.image, img.x, img.y, img.width, img.height, -(px + width), py, width, height);
366 							this.context.scale(-1, 1);
367 						}
368 					}
369 				}
370 			} else if (layer instanceof SpriteLayer) {
371 				var count = layer.count;
372 				for (i = 0; i < count; i ++) {
373 					offset = 4 * i;
374 					var minX = layer.rectangles[offset ++],
375 						minY = layer.rectangles[offset ++],
376 						maxX = layer.rectangles[offset ++],
377 						maxY = layer.rectangles[offset ++];
378 
379 					if (maxX >= x && minX <= x + nx && maxY >= y && minY <= y + ny) {
380 						if (this.calls++ > this.maxCalls) continue;
381 						offset = 5 * i;
382 						img = layer.tiles[offset ++];
383 						ix = layer.tiles[offset ++];
384 						iy = layer.tiles[offset ++];
385 						centered = layer.tiles[offset ++];
386 						flipped = layer.tiles[offset ++];
387 
388 						this.drawImage(img, ox + ix, oy + iy, 0, centered, 1, null, flipped);
389 					}
390 				}
391 			} else {
392 				assert(false);
393 			}
394 		},
