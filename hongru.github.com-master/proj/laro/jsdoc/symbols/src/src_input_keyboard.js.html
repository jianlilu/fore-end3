  6 // 保证一个 keyboard 的事件注册在一个游戏里面只注册一次
  7 Laro.register('.input', function (La) {
  8 	var pkg = this;
  9 	var keyHash = {
 10 		'a': 65,

 52 		'enter': 13,
 53 
 54 		'~': 192,
 55 		'-': 189,

 73 		'left': 37,
 74 		'right': 39,
 75 
 76 		'shiftKey': 16,
 77 		'ctrlKey': 17,
 78 		'altKey': 18
 79 	};
 80 	
 81 	var Keyboard = La.Class(function (canvas) {
 82 		this.target = canvas || window || window.document;
 83 		this.keyHash = keyHash;
 84 		this.keyStatus = {};
 85 		this.keyStack = [];
 86 		
 87 		this.keydownCallbacks = {};
 88 		this.keyupCallbacks = {};
 89 		
 90 		this._downKey = null;
 91 		this._upKey = null;
 92 		this._lastKeyupTime = (+new Date);
 93 		this.checkTime = 200;
 94 		this.maxStackLength = 10;
 95 		this.bind();
 96 	}).methods({
 97 		bind: function () {
 98 			var _this = this;
 99 			this.target.addEventListener('keydown', function (e) {
100 				if (e.keyCode == 32 || e.keyCode == 37 || e.keyCode == 38 || e.keyCode == 39 || e.keyCode == 40) {
101 					if (e.preventDefault) {
102 						e.preventDefault();
103 					} else {
104 						e.returnValue = false;
105 					}
106 				}
107 				
108 				if (!e.$keyName) {
109 					e.$keyName = _this.getKeyName(e.keyCode);
110 				}
111 				
112 				_this.keydown(e);
113 				for (var i in _this.keydownCallbacks) {
114 					var fn = _this.keydownCallbacks[i];
115 					if (typeof fn == 'function') {
116 						fn(e);
117 					}
118 				}
119 			}, false);
120 			this.target.addEventListener('keyup', function (e) { 
121 				if (e.keyCode == 32 || e.keyCode == 37 || e.keyCode == 38 || e.keyCode == 39 || e.keyCode == 40) {
122 					if (e.preventDefault) {
123 						e.preventDefault();
124 					} else {
125 						e.returnValue = false;
126 					}
127 				}
128 				
129 
130 				_this.keyup(e);
131 				for (var i in _this.keyupCallbacks) {
132 					var fn = _this.keyupCallbacks[i];
133 					typeof fn == 'function ' && fn(e);
134 				}
135 			}, false);
136 		},
137 		addKeydownCallback: function (name, cb) {
138 			this.keydownCallbacks[name] = cb;
139 		},
140 		addKeyupCallback: function (name, cb) {
141 			this.keyupCallbacks[name] = cb;
142 		},
143 		removeKeydownCallback: function (name) {
144 			delete this.keydownCallbacks[name];
145 		},
146 		removeKeyupCallback: function (name) {
147 			delete this.keyupCallbacks[name];
148 		},
149 		getKeyName: function (keyCode) {
150 			for (var name in keyHash) {
151 				if (keyHash[name] == keyCode) {
152 					return name;
153 				}
154 			}
155 			return keyCode;
156 		},
157 		keydown: function (e) {
158 			if (this.keyStack.length == 0 || (+new Date) - this._lastKeyupTime > this.checkTime) {
159 				this.resetKeyStack();
160 			}
161 			var name = this.getKeyName(e.keyCode);
162 			!this.key(name) && this.pushToStack(name);
163 			this.keyStatus[name] = true;
164 			this._downKey = name;
165 		},
166 		keyup: function (e) {
167 			var name = this.getKeyName(e.keyCode);
168 			this.keyStatus[name] = false;
169 			this._upKey = name;
170 
171 			//this.updateKeyStack();
172 		},
173 		/**
174 		 * 判断某个键是否按下
175 		 * 按下为true, 否则false
176 		 */
177 		key: function (name) {
178 			return !!this.keyStatus[name];	 
179 		},
180 		// 更新keyStack,主要是用于一些组合键的判断
181 		pushToStack: function (name) {
182 			var t = (+new Date);
183 			if (t - this._lastKeyupTime > this.checkTime) {
184 				this.resetKeyStack(t);
185 				return;
186 			}
187 			this._lastKeyupTime = t;
188 			this.keyStack.push(name);
189 			var l = this.keyStack.length,
190 				tmpStack = this.keyStack;
191 			if (l > this.maxStackLength) {
192 				this.keyStack = tmpStack.splice(l - this.maxStackLength);
193 			}
194 		},
195 		resetKeyStack: function (t) {
196 			this.keyStack = [];
197 			this._lastKeyupTime = t || (+new Date);
198 		}
199 		
200 	}).statics({
201 		
202 		
203 	});
204 
205 	this.Keyboard = Keyboard;
206 	La.extend({ Keyboard: Keyboard });
207 		
208 })
