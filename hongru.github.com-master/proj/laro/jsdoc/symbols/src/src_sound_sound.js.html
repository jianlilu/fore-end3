1 /**
  2  * Sound pkg
  3  * base on html5 audio
  4  */
  5  
  6 Laro.register('.game', function (La) {
  7 	
  8 	var Sound = La.Class(function (url, callback, isAuto) {
  9 		this.url = url;
 10 		this.loaded = false;
 11 		this.callback = callback;
 12 		this.channels = {};
 13 		this.currentChannel = null;
 14 		this.genAudio();
 15 		
 16 		this.can = this.canPlayThisType();
 17 		if (!this.can) {
 18 			!!this.callback && this.callback();
 19 			return;
 20 		}
 21 		
 22 		this.bind();
 23 		this.audio.load();
 24 			
 25 		
 26 	}).methods({
 27 		genAudio: function () {
 28 			this.audio = new Audio();
 29 			this.audio.src = this.url;
 30 			this.audio.preload = 'auto';
 31 		
 32 			if (this.audio.play && navigator.userAgent.toLowerCase().indexOf('msie') < 0) {
 33 				this.audio.play();
 34 				this.audio.pause();
 35 			}
 36 
 37 		},
 38 		canPlayThisType: function (type) {
 39 			var checkTypeHash = {
 40 				'mp3': 'audio/mpeg;',
 41 				'm4a': 'audio/aac',
 42 				'ogg': 'audio/ogg; codecs="vorbis"',
 43 				'wav': 'audio/wav'
 44 			};
 45 			if (type == undefined) {
 46 				type = this.url.substr(this.url.lastIndexOf('.')).replace(/\./g, '');
 47 			}
 48 			
 49 			return !!(this.audio && this.audio.canPlayType && this.audio.canPlayType(checkTypeHash[type]).replace(/no/, ''));
 50 		},
 51 		bind: function () {
 52 			var _this = this;
 53 			this.audio.addEventListener('loadedmetadata', function (e) { 
 54 				_this.addChannel('default', 0, _this.getDuration());	
 55 			}, false);
 56 			this.audio.addEventListener('playing', function(e) { 
 57 				// todo	
 58 			}, false);
 59 			this.audio.addEventListener('timeupdate', function (e) {
 60 				// todo
 61 			}, false);
 62 			this.audio.addEventListener('canplaythrough', function (e) { 
 63 				_this.loaded = true;
 64 				!_this.channels['default'] && _this.addChannel('default', 0, _this.getDuration());
 65 				!!_this.callback && _this.callback();
 66 			}, false);
 67 		},
 68 		update: function (dt) {
 69 			// todo
 70 		},
 71 		getVolume: function () {
 72 			return this.audio.volume;		   
 73 		},
 74 		setVolume: function (v) {
 75 			v = Math.max(0, Math.min(v, 100));
 76 			this.audio.volume = v;		   
 77 		},
 78 		haveData: function () {
 79 			var s = this.audio.readyState;
 80 			if (s == 1 || !s) {
 81 				return false;
 82 			} else if (s == 2 || s == 3 || s == 4 || s == 5) {
 83 				return true;
 84 			}
 85 		},
 86 		isPlayingChannel: function (name) {
 87 			return !this.audio.paused && (name === this.currentChannel.name);				  
 88 		},
 89 		getDuration: function () {
 90 			return this.audio.duration;
 91 		},
 92 		getCurrentChannel: function () {
 93 			return this.currentChannel;
 94 		},
 95 		addChannel: function (name, start, duration) {
 96 			if (La.toType(name) == 'object') {
 97 				name = name.name;
 98 				start = name.start;
 99 				duration = name.duration;
100 			}
101 			var wholeDuration = this.getDuration();
102 			if (start > wholeDuration) {
103 				throw 'Sound channel start wrong!'
104 			}
105 
106 			if (start + duration > wholeDuration) {
107 				duration = wholeDuration - start;
108 			}
109 			this.channels[name] = {
110 				start: start,
111 				end: (start + duration),
112 				duration: duration
113 			};
114 			
115 			return this.channels[name];
116 		},
117 		removeChannel: function (name) {
118 			delete this.channels[name];			   
119 		},
120 		pause: function () {
121 			return this.audio.pause();	   
122 		},
123 		preloadChannel: function (name) {
124 			this.pause();
125 			var channel = this.channels[name];
126 			if (channel) {
127 				this.audio.currentTime = channel.start;
128 			}
129 		},
130 		play: function (name, isLoop) {
131 			if (!this.can) { return; }
132 			
133 			var _this = this, timer = -1;
134 			
135 			if (name == undefined) {
136 				name = 'default';
137 			}
138 			if (isLoop == undefined) {
139 				isLoop = false;
140 			}
141 			
142 			if (this.currentChannel) {
143 				clearTimeout(this.currentChannel.timer);
144 			}
145 			if (this.haveData()) {
146 				var channel = this.channels[name];
147 				clearTimeout(channel.timer);
148 				this.audio.currentTime = channel.start;
149 				this.audio.play();
150 				
151 				if (isLoop) {
152 					timer = setTimeout(function () {
153 						_this.play(name, isLoop)
154 					}, channel.duration*1000);
155 				}
156 				
157 				
158 				this.currentChannel = {
159 					name: name,
160 					start: channel.start,
161 					duration: channel.duration,
162 					end: channel.end,
163 					isLoop: isLoop,
164 					startTime: (+new Date),
165 					timer: timer
166 				};
167 			}
168 		}
169 	}).statics({
170 		CAN_PLAY_TYPES: function () {
171 			var a = document.createElement('audio'), o = {}, s = "audio\/";
172 			if (typeof a.canPlayType === "function") {
173         			o._supported = Boolean(true);
174         			o.mp3 = a.canPlayType(s + "mpeg");
175         			o.wav = a.canPlayType(s + 'wav; codecs="1"');
176         			o.ogg = a.canPlayType(s + 'ogg; codecs="vorbis"');
177         			o.m4a = a.canPlayType(s + "x-m4a") || a.canPlayType(s + "aac");
178         			o.webm = a.canPlayType(s + 'webm; codecs="vorbis"');
179     			} else {
180         			o._supported = Boolean(0);
181     			}
182     			
183 			return o;
184 		}()
185 	})
186 
187 	this.Sound = Sound;
188 	La.extend({Sound: Sound});
189 
190 })
