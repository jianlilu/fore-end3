 17 Laro.register('.action', function (La) {
 18 
 19 	var Class = La.Class || La.base.Class,
 20 		extend = La.extend;
 21 
 22 	/**
 23      * 帧动画信息 类
 24      * @class 
 25 
 26 	 * @memberOf Laro
 27      * @name Animation
 28      * @constructor
 29      * 
 30      * @param {Object} anim: 从json配置里面获取的anim 配置
 31 			{
 32 				"nbrOfFrames": 73,
 33 				"name": "TimeTrap",
 34 				"atlas": "atlas/game/timetrap",
 35 				"type": "animation",
 36 				"image": "anims/timetrap.png",
 37 				"pivoty": 128,
 38 				"framerate": 30,
 39 				"pivotx": 256,
 40 				"events": []
 41 			}
 42      * @param {Array} frames: 从json配置文件里面获得的每帧的位置信息
 43      * @return 帧动画信息实例
 44      */
 45 
 46 	var Animation = Class(function (anim, frames) {
 47 		extend(this, anim);
 48 
 49 		this.frames = frames;
 50 		if (anim.framerate == undefined) anim.framerate = 20;
 51 		// 这个动画执行需要的时间
 52 		this.animationLength = frames.length / anim.framerate;
 53 	}).methods({
 54 	/**
 55      * @lends Laro.Animation.prototype
 56      */ 
 57 
 58 		/**
 59 		 * 获取动画时间内指定时间段[from, to]中插入的事件
 60 		 * @param {Number} from: 起始时间
 61 		 * @param {Number} to: 结束时间
 62 		 * @return {Array} 指定时间区间内事件列表
 63 		 */
 64 		getEvents: function (from, to) {
 65 			var events = [];
 66 			for (var e = 0; e < this.events.length; e ++) {
 67 				var evt = this.events[e];
 68 				if (evt.time >= from && evt.time < to) {
 69 					events.push(evt.name);
 70 				}
 71 			}
 72 			return events;
 73 		},
 74 		/**
 75 		 * 获取下一个动画内（指定时间段内）插入事件的触发具体时间
 76 		 * @param {Number} from: 起始时间
 77 		 * @param {Number} to: 结束时间
 78 		 * @return {Number} 插入事件的触发具体时间点
 79 		 */
 80 		getTimeForNextEvent: function (from, to) {
 81 			var first = -1;
 82 			for (var e = 0; e < this.events.length; e ++) {
 83 				var evt = this.events[e];
 84 				if (evt.time > from && evt.time < to) {
 85 					if (first != -1) return first;
 86 					first = evt.time;
 87 				}
 88 			}
 89 			return first;
 90 		},
 91 		/**
 92 		 * 给定两个时间区间，如果有交集，交集中的事件push两次
 93 		 */ 
 94 		getEventsSlow: function (from, to, start, end, dt) {
 95 			var events = [],
 96 				e,
 97 				evt;
 98 			for (e = 0; e < this.events.length; e++) {
 99 				evt = this.events[e];
100 				if (evt.time >= from && evt.time < end) {
101 					events.push(evt.name);
102 				}
103 			}
104 
105 			for (e = 0; e < this.events.length; e ++) {
106 				evt = this.events[e];
107 				if (evt.time >= start && evt.time < to) {
108 					events.push(evt.name);
109 				}
110 			}
111 
112 			return events;
113 		}
114 	});
115 
116 	this.Animation = Animation;
117 	Laro.extend(this);
118 		
119 })
