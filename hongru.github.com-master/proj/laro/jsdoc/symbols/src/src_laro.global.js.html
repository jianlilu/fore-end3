 13 (function (win, undefined) {
 14     /**
 15      * Laro - 顶级命名空间

 17      * @namespace
 18      * @name Laro

 20  	var __INFO__ = {
 21 		$name: 'Laro',
 22 		$version: '0.1',
 23 		$description: 'game engine based on html5'
 24 	};

 26 	var	toString = Object.prototype.toString,
 27 		slice = Array.prototype.slice,
 28 		self = this || win;

 30 	/**
 31      * 判断指定参数的 type
 32      * 
 33      * @memberOf Laro
 34      * @function
 35      * 
 36      * @param {*} 需要接受类型判断的参数
 37      * 
 38      * @return {String}指定参数的类型
 39      */
 40 	function toType (o) {
 41 		var r = toString.call(o).toLowerCase(),
 42 			from = 8,
 43 			to = r.length - 1;
 44 		return r.substring(from, to);
 45 	}

 47 	/**
 48      * 扩展一个指定的对象的属性 或 方法，如果只有一个参数，表示直接扩展顶级命名空间 Laro
 49      * 
 50      * @memberOf Laro
 51      * @function
 52      * 
 53      * @param {Object} target 被扩展的对象
 54      * @param {Object} source 用于扩展的源对象
 55      * @param {Boolean} isOverwrite 指定是否覆盖已有属性或方法
 56 	 *
 57 	 * @return {Object} obj 扩展之后的目标对象
 58      * 
 59      */
 60 	function extend (target, source, isOverwrite) {
 61 		var argInd = -1,
 62 			args = slice.call(arguments, 0);
 63 		target = self[__INFO__['$name']] || {};
 64 		source = [];
 65 		isOverwrite = true;
 66 		while (args[++ argInd]) {
 67 			if (toType(args[argInd]) === 'boolean') {
 68 				isOverwrite = args[argInd];
 69 			} else if (toType(args[argInd]) === 'object') {
 70 				source.push(args[argInd]);
 71 			} 
 72 		}
 73 
 74 		if (source.length >= 2) {
 75 			target = source.splice(0, 1)[0];
 76 		}
 77 
 78 		for (var i = 0; i < source.length; i ++) {
 79 			var _s = source[i];
 80 			for (var key in _s) {
 81 				if (!target.hasOwnProperty(key) || isOverwrite) {
 82 					target[key] = _s[key];
 83 				}
 84 			}
 85 		}
 86 
 87 		return target;
 88 	}

 90 	/**
 91      * 注册一个命名空间
 92      * 
 93      * @memberOf Laro
 94      * @function
 95      * 
 96      * @param {String} name: 注册的命名空间名字，以‘.’开始表示注册为Laro的子命名空间，如 Laro.register('.test', function () {...}) ; 否则表示注册顶级命名空间
 97      * @param {Function} fn: 命名空间属性或方法的 提供 者
 98      * 
 99      */
100 	function register (name, fn) {
101 		var names = name.split('.'),
102 			i = -1,
103 			loopName = self;
104 
105 		if (names[0] == '') {names[0] = __INFO__['$name']}
106 
107 		while (names[++ i]) {
108 			if (loopName[names[i]] === undefined) {
109 				loopName[names[i]] = {};
110 			}
111 			loopName = loopName[names[i]]
112 		}
113 
114 		!!fn && fn.call(loopName, self[__INFO__['$name']]);
115 		
116 	}

118 	/**
119      * 返回指定区间的随机数
120      * 
121      * @memberOf Laro
122      * @function
123      * 
124      * @param {Number} from: 区间开始
125      * @param {Number} to: 区间结束
126 	 *
127 	 * @return {Number} num 指定区间的随机数
128      * 
129      */
130 	function randomRange(from, to) {
131 		return from + Math.random() * (to - from);
132 	}
133 	
134 	/**
135      * 返回随机的 true 或者 false[相同概率]
136      * 
137      * @memberOf Laro
138      * @function
139 	 *
140 	 * @return {Boolean} bool 相同概率的随机true或者 false
141      * 
142      */
143 	function randomBool() {
144 		return Math.random() >= 0.5;
145 	}
146 	/**
147      * 科里化
148      * 
149      * @memberOf Laro
150      * @function
151      * 
152      * @param {Function} cb: 被科里化的方法
153      * @param {Object} context: 指定上下文对象
154 	 *
155 	 * @return {Function}  科里化之后的方法
156      * 
157      */
158 	function curry (cb, context) {
159 		return function () {
160 			typeof cb == 'function' && cb.apply(context, arguments);
161 		}
162 	}
163 	
164 	/**
165      * 带参数的科里化
166      * 
167      * @memberOf Laro
168      * @function
169      * 
170      * @param {Function} cb: 被科里化的方法
171      * @param {Object} context: 指定上下文对象
172      * @param {*} 科里化后的方法调用时参数
173 	 *
174 	 * @return {Function} fun 科里化之后的方法
175      * 
176      */
177 	function curryWithArgs(cb, context) {
178 		var args = Array.prototype.slice.call(arguments, 0);
179 		delete args[0];
180 		delete args[1];
181 		return function () {
182 			typeof cb == 'function' && cb.apply(context, args.concat(arguments));
183 		};
184 	}
185 
186 	var $public = {
187 		toType: toType,
188 		extend: extend,
189 		register: register,
190 		randomRange: randomRange,
191 		randomBool: randomBool,
192 		curry: curry,
193 		curryWithArgs: curryWithArgs
194 	};
195 
196 
197 	var Laro = extend({}, __INFO__, $public);
198 	this[__INFO__['$name']] = win[__INFO__['$name']] = Laro;
199  
200  })(window);
