 62 	/**
 63 	 * ConvexShape
 64 	 * 凸起物体基类
 65 	 * @param {String} type - 'rect' | 'circle' | 'poly'
 66 	 * @param {Number} mat 质地
 67 	 */
 68 	var ConvexShape = La.Class(function (type, mat) {
 69 		this.type = type;
 70 		if (mat == undefined) {
 71 			mat = 0;
 72 		}
 73 		this.material = mat;
 74 		this.y_min = Number.MAX_VALUE;
 75 		this.user = null;
 76 		this.points = [];
 77 			
 78 	}).methods({
 79 		//增加一个顶点 Vec2
 80 		addPoint: function (v) {
 81 			this.points.push(v);
 82 			if (v.y < this.y_min) {
 83 				this.y_min = v.y;
 84 			}
 85 		},
 86 		//清除 顶点
 87 		clear: function () {
 88 			this.points = [];
 89 			this.y_min = Number.MAX_VALUE;
 90 			return this;
 91 		},
 92 		numOfPoints: function () {
 93 			return this.points.length;
 94 		},
 95 		//移动指定 向量
 96 		offset: function (v) {
 97 			this.y_min += v.y;
 98 			for (var i = 0; i < this.points.length; i ++) {
 99 				this.points[i].add(v);
100 			}
101 
102 			return this;
103 		},
104 		// 横向翻转
105 		// @param {Number} x ,翻转的x坐标
106 		hFlip: function (x) {
107 			this.points.reverse();
108 			for (var i = 0; i < this.points.length; i ++) {
109 				this.points[i].x = x - (this.points[i].x - x);
110 			}
111 
112 			return this;
113 		},
114 		// 生成一个新的拷贝
115 		copy: function () {
116 			var c = new this.ConvexShape(this.type, this.material);
117 			for (var i = 0; i < this.points.length; i ++) {
118 				c.addPoint(this.points[i].copy());
119 			}
120 
121 			return c;
122 		}
123 		
124 	});

128 	/**
129 	 * 二维线段
130 	 * LineSegment2
131 	 * @param {Point2} start
132 	 * @param {Point2} end
133 	 */
134 	var LineSegment2 = La.Class(function (start, end) {
135 		this.a = new Vec2(start.x, start.y);
136 		this.b = new Vec2(end.x, end.y);
137 			
138 	}).methods({
139 		// @param {Vec2} p
140 		// @return {Vec2}
141 		// 获取线段上 到指定点p 最短距离的点
142 		getClosestPoint: function (p) {
143 			var val = this.getProjectionParam(p);
144 			// Clamp
145 			var param = val < 0 ? 0 : val > 1 ? 1 : val;
146 		 
147 			var res = new Vec2(this.b.x, this.b.y);
148 			res.sub(this.a);
149 			res.mul(param);
150 			res.add(this.a);
151 		 
152 			return res;
153 		},
154 		// @param {Vec2} p
155 		// @return {number}
156 		// 获取距离p点最短距离的点 的 比例
157 		getProjectionParam: function (p) {
158 			var m = this.b.copy().sub(this.a); // m = b - a
159 			var proj = p.copy().sub(this.a); // p - a
160 			return m.dot(proj) / m.dot(m); // m.(p-a) / m.m
161 		},
162 		// @param {Vec2} p
163 		// @return {number}
164 		// 得到 p 点到线段的最短距离
165 		getProjectionDistance: function (p) {
166 			var _p = new Vec2(p.x, p.y);
167 			_p.sub(this.getClosestPoint(p));
168 			return _p.magnitude();
169 		}
170 		
171 	});

174 	/* == Helper == */
175 	// 用于帮助检测 碰撞
176 	/**
177 	 * Helper area calculation function. Returns 2 X the area.
178 	 * 三角形外积
179 	 *
180 	 * @param  {Vec2} pointA
181 	 * @param  {Vec2} pointB
182 	 * @param  {Vec2} pointC
183 	 * @return {number}
184 	 */
185 	function signed2DTriArea(pointA, pointB, pointC) {
186 		 return ((pointA.x - pointC.x) * (pointB.y - pointC.y) - (pointA.y - pointC.y) * (pointB.x - pointC.x));
187 	}
188 	this.signed2DTriArea = signed2DTriArea;

190 	/**
191 	 * Helper intersection function. Intersects a ray with a sphere.
192 	 *
193 	 * @param {Ray2}   ray      Ray for intersecting
194 	 * @param {Circle} circle   Circle to be intersected
195 	 * @return {boolean}        If the ray intersects the sphere
196 	 * 判断 射线与 圆 是否相交
197 	 */
198 	var closure;
199 	function intersectRaySphere(ray, circle) {
200 		closure = function () {
201 			return localCollisionTime;   
202 		};
203 	 
204 		var m = new Vec2(ray.point.x, ray.point.y);
205 		m.sub(circle.c);
206 	 
207 		var a = ray.normal.dot(ray.normal);
208 		var b = m.dot(ray.normal);
209 		var c = m.dot(m) - circle.r * circle.r;
210 	 
211 		// Is the ray heading towards the circle? Otherwise it won't intersect
212 		if (c > 0 && b > 0) {
213 			return false;
214 		}
215 		
216 		var discr = b * b - a * c;
217 	 
218 		// Do the ray intersect the sphere if the ray was infinitely long?
219 		if (discr < 0) {
220 			return false;
221 		}
222 		
223 		var localCollisionTime = (-b - Math.sqrt(discr)) / a;
224 	 
225 		// Check that the collision time is on the ray
226 		if (localCollisionTime < 0 || localCollisionTime > 1) {
227 			return false;
228 		}
229 	 
230 		return true;
231 	}
232 	this.intersectRaySphere = intersectRaySphere;
233 	
234 	/**
235 	 * Helper intersection function. Checks if two lines intersect.
236 	 *
237 	 * @param {LineSegment2} a Line A
238 	 * @param {LineSegment2} b Line B
239 	 * @return {Object} An object is returned with intersection point and time if they intersect, otherwise null.
240 	 * 判断两条线段是否相交，如果是，返回交点，和相交时间， 否则返回null
241 	 */
242 	function intersectLineSegments(a, b) {
243 		var a1 = signed2DTriArea(a.a, a.b, b.b);
244 		var a2 = signed2DTriArea(a.a, a.b, b.a);
245 		if (a1 * a2 < 0) {
246 			var a3 = signed2DTriArea(b.a, b.b, a.a);
247 			var a4 = a3 + a2 - a1;
248 			if (a3 * a4 < 0) {
249 				var intersectionTime = a3 / (a3 - a4);
250 	 
251 				// intersectionPoint = a.a + intersectionTime * (a.b - a.a);
252 				var intersectionPoint = new Vec2(a.b.x, a.b.y);
253 				intersectionPoint.sub(a.a);
254 				intersectionPoint.mul(intersectionTime);
255 				intersectionPoint.add(a.a);
256 	 
257 				return {intersectionPoint: intersectionPoint,intersectionTime : intersectionTime};
258 			}
259 		}
260 	 
261 		return null;
262 	}
263 	this.intersectLineSegments = intersectLineSegments;

278 	/**
279 	 * Test a sweep-circle against convex shape.
280 	 *
281 	 * @param {Circle}      circle        Circle used in test
282 	 * @param {Vec2}        movement      Vector describing movement
283 	 * @param {ConvexShape} shape         Shape to be tested against
284 	 * @return {Collisions}               Collision objects that holds both contact points and intersection time.
285 	 * 判断一个移动的圆 和 不规则凸起形状的碰撞相交关系，如果相交，返回关联的点和 相交时间
286 	 * convex shape 的 points 顶点顺序 需要是逆时针排列的，这样可以避免在内部碰撞的检测，同时movement反向的也可以直接跳出
287 	 */
288 	function getCollisionShape(circle, movement, shape) {
289 		var contactPoints = [],
290 			SWEEP_EPSILON = pkg.SWEEP_EPSILON;
291 	 
292 		var foundOne = false;
293 		var minIntersectionTime = Number.MAX_VALUE;
294 		var localIntersectionTime = Number.MAX_VALUE;
295 	 
296 		var i = 0;      // Used for iteration
297 		var pt;         // Handle during loops
298 		var contact;    // Placeholder for eventual contact point
299 	 
300 		if (shape.numOfPoints() > 1) {
301 			pt = shape.points;
302 			for (i = 0; i < pt.length; i++) {
303 				var last = pt[i];
304 				var curr = i+1 === pt.length ? pt[0] : pt[i+1];
305 	 
306 				// last & curr is now start and end points of the line.
307 				// 远离
308 				var normal = new Vec2(-(curr.y - last.y), curr.x - last.x);
309 				if (0 < normal.dot(movement)) {
310 					continue; 
311 				}
312 	 
313 				normal.normalize();
314 	 
315 				// last = last + normal * (circle.r + SWEEP_EPSILON)
316 				var _last = new Vec2(normal.x, normal.y);
317 				_last.mul(circle.r + SWEEP_EPSILON);
318 				_last.add(last);
319 	 
320 				// curr = curr + normal * (circle.r + SWEEP_EPSILON)
321 				var _curr = new Vec2(normal.x, normal.y);
322 				_curr.mul(circle.r + SWEEP_EPSILON);
323 				_curr.add(curr);
324 	 
325 				var endPos = new Vec2(circle.c.x, circle.c.y);
326 				endPos.add(movement);
327 	 
328 				var localContact = new Vec2();
329 	 
330 				var res = intersectLineSegments(new LineSegment2(circle.c, endPos), new LineSegment2(_last, _curr));
331 				if (res) {
332 					if (res.intersectionTime < minIntersectionTime) {
333 						foundOne = true;
334 						minIntersectionTime = res.intersectionTime;
335 	 
336 						// localContact - normal * circle.r
337 						var _lc = new Vec2(normal.x, normal.y);
338 						_lc.mul(circle.r);
339 	 
340 						res.intersectionPoint.sub(_lc);
341 	 
342 						contact = new CollisionContact(res.intersectionPoint, normal, 0, shape.user, shape.material);
343 					}
344 				}
345 			}
346 		}
347 	 
348 	 
349 		if (!foundOne) {
350 			pt = shape.points;
351 			for (i = 0; i < pt.length; i++) {
352 				localIntersectionTime = Number.MAX_VALUE; 
353 	 
354 				// Multiply the length of the ray to make sure that the spheres won't go
355 				// through each other at extremely low speeds
356 				var _mov = new Vec2(movement.x, movement.y);
357 				_mov.mul(1.1);
358 				if (intersectRaySphere(new Ray2(circle.c, _mov), new Circle(pt[i], circle.r))) {
359 					localIntersectionTime = closure();
360 					if (localIntersectionTime < minIntersectionTime) {
361 						// Got collision.
362 						foundOne = true;
363 						minIntersectionTime = localIntersectionTime;
364 						
365 						// circle.c + localIntersectionTime * movement - pt[i]
366 						var _normal = new Vec2(movement.x, movement.y);
367 						_normal.mul(localIntersectionTime);
368 						_normal.sub(pt[i]);
369 						_normal.add(circle.c);
370 						_normal.normalize();
371 	 
372 						// Contact point is the vertex.
373 						// Normal is vector from corner to position of sphere at collision time.
374 						contact = new CollisionContact(pt[i], _normal, 0, shape.user, shape.material);
375 					}
376 				}
377 			}
378 		}
379 		
380 	 
381 		if (foundOne) {
382 			contactPoints.push(contact);
383 			return new Collisions(contactPoints, minIntersectionTime);
384 		}
385 		else {
386 			return null;
387 		}
388 	};
389 	this.getCollisionShape = getCollisionShape;

391 	/**
392 	 * Test two sweep circles against eachother
393 	 *
394 	 * @param {Circle} circleA
395 	 * @param {Vec2}   movementA
396 	 * @param {Circle} circleB
397 	 * @param {Vec2}   movementB
398 	 * @return {Collisions}  Collision objects that holds both contact points and intersection time.
399 	 * 两个移动的圆的 相交 关系
400 	 */
401 	function getCollisionSphere(circleA, movementA, circleB, movementB) {
402 		// Calculating the sum of the movement of the two circles. The test is reduced
403 		// to a ray test against a stationary circle
404 		
405 		// movementA - movementB
406 		var movementOfCircleA = new Vec2(movementA.x, movementA.y);
407 		movementOfCircleA.sub(movementB);
408 		var cB = new Circle(circleB.c, circleA.r + circleB.r);
409 	 
410 		// Now do a regular ray-circle intersection.
411 		var contactPoints = [];
412 		var collision = null;
413 		
414 		// Multiply the length of the ray with 1.1 to make sure that the spheres won't go
415 		// through each other at extremely low speeds
416 		if (intersectRaySphere(new Ray2(circleA.c, movementOfCircleA.mul(1.1)), cB)) {
417 			var localCollisionTime = closure();
418 	 
419 			var contactPoint = new CollisionContact();
420 	 
421 			// Normal is line between centers at collision time.
422 			// contactPoint.normal = ((circleA.c + localCollisionTime * movementA) - (circleB.c + localCollisionTime * movementB)).normalize();
423 			var rhs = new Vec2(movementB.x, movementB.y); // RHS of -
424 			rhs.mul(localCollisionTime);
425 			rhs.add(circleB.c);
426 	 
427 			contactPoint.normal = new Vec2(movementA.x, movementA.y); // LHS of -
428 			contactPoint.normal.mul(localCollisionTime);
429 			contactPoint.normal.add(circleA.c);
430 	 
431 			contactPoint.normal.sub(rhs);
432 			contactPoint.normal.normalize();
433 	 
434 			// Collision point.
435 			// contactPoint.point = (circleB.c + localCollisionTime * movementB) + contactPoint.normal * circleB.r;
436 			rhs = new Vec2(contactPoint.normal.x, contactPoint.normal.y); // RHS of outer +
437 			rhs.mul(circleB.r);
438 	 
439 			contactPoint.point = new Vec2(movementB.x, movementB.y); // LHS of outer +
440 			contactPoint.point.mul(localCollisionTime);
441 			contactPoint.point.add(circleB.c);
442 	 
443 			contactPoint.point.add(rhs);
444 	 
445 			contactPoints.push(contactPoint);
446 			collision = new Collisions(contactPoints, localCollisionTime);
447 		}
448 	 
449 		return collision;
450 	};
451 	this.getCollisionSphere = getCollisionSphere;

453 	/**
454 	 * Test if a sphere is penetrating a convex shape.
455 	 *
456 	 * @param {Circle}              circle
457 	 * @param {Vec2}                pos [偏移向量pos]
458 	 * @param [ConvexShape]         shapes
459 	 *
460 	 * @return {[CollisionContact]}  contactPoints
461 	 * 圆是否穿过 某个凸起物
462 	 */
463 	function getPenetrations(circle, pos, shapes) { 
464 		var contactPoints = [],
465 			PENETRATION_EPSILON = pkg.PENETRATION_EPSILON,
466 			SWEEP_EPSILON = pkg.SWEEP_EPSILON;
467 			
468 		for (var i = 0; i < shapes.length; i++) {
469 			if (shapes[i].numOfPoints() > 1) {
470 				for (var p = 0; p < shapes[i].points.length; p++) {
471 					var last = shapes[i].points[p];
472 					var curr = p+1 === shapes[i].points.length ? shapes[i].points[0] : shapes[i].points[p+1];
473 	 
474 					var circlePos = new Vec2(circle.c.x, circle.c.y);
475 					circlePos.add(pos);
476 	 
477 					var ls = new LineSegment2(curr, last);
478 					var closestPoint = ls.getClosestPoint(circlePos);
479 	 
480 					var contactVector = new Vec2(closestPoint.x, closestPoint.y);
481 					contactVector.sub(circlePos);
482 
483 					if (contactVector.dot(contactVector) <= ((circle.r + PENETRATION_EPSILON) * (circle.r + PENETRATION_EPSILON))) { 
484 						var depth = (circle.r + SWEEP_EPSILON) - contactVector.magnitude();
485 	 
486 						// faceNormal = last - curr
487 						var faceNormal = new Vec2(last.x, last.y);
488 						faceNormal.sub(curr);
489 						faceNormal.y *= -1;
490 						faceNormal.normalize();
491 	 
492 						contactVector.mul(-1);
493 	 
494 						var contact = new CollisionContact(closestPoint, contactVector, depth, faceNormal, shapes[i].user, shapes[i].material);
495 						contact.normal.normalize();
496 	 
497 						contactPoints.push(contact);
498 					}
499 				}
500 			}
501 		}
502 	 
503 		return contactPoints;
504 	};
505 	this.getPenetrations = getPenetrations;
